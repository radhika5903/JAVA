import java.util.Scanner;
public class Main 
 {
    public static void main(String[] args)
    {
        Scanner sc = new Scanner(System.in);
        int t = sc.nextInt();
        if (t <= 10 || t> 21) {
            System.out.println(0);
        } else if (t == 20) {
            System.out.println(15);
        } else {
            System.out.println(4);
        }
    }
}
